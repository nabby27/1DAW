Algoritmo: calcularSalario

Real: horas, salario
Constantes: precioHora, horasSemanales

INICIO
    precioHora = 20
    horasSemanales = 40
    escribir("Dime el numero de horas semanales")
    leer horas
    mientras (horas - horasSemanales > horasSemanales) entonces
        salario = horasSemanales * precioHora;
        horas = horas - horasSemanales
        salario = salario + (horas * precioHora * 1,5) 
    finMinetras
        salario = salario + (horas * precioHora)
FIN_ALGORITMO
